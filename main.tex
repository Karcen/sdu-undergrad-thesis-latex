% ================================================================
%  19. 图/表/代码目录（居中标题 + 双语支持）
% ================================================================
% 说明：
%   · \listoffigures 等原版命令在 ctex section 的 \raggedright 格式下
%     右侧 \hfill 失效，导致标题左对齐；此处用 center 环境强制居中。
%   · 新增 \listoffiguresEN / \listoftablesEN / \lstlistoflistingsEN，
%     与中文版共享同一辅助文件（.lof / .lot / .lol）。
\makeatletter

% 内部命令：生成居中的列表大标题并写入 ToC
%   #1  写入正文目录的条目文字
%   #2  页面上居中显示的标题
%   #3  列表辅助文件后缀（lof / lot / lol）
\newcommand\@centeredlisttitle[3]{%
  \clearpage
  \phantomsection
  \addcontentsline{toc}{section}{#1}%
  \begingroup
    \par\vspace*{0.4\baselineskip}%
    \begin{center}
      \heiti\zihao{-2}\bfseries #2
    \end{center}
    \vspace{0.4\baselineskip}%
  \endgroup
  \@starttoc{#3}%
}

% 中文版
\renewcommand\listoffigures{%
  \@centeredlisttitle{图\quad 目\quad 录}{图\qquad 目\qquad 录}{lof}}
\renewcommand\listoftables{%
  \@centeredlisttitle{表\quad 目\quad 录}{表\qquad 目\qquad 录}{lot}}

\renewcommand\lstlistingname{代码}
\renewcommand\lstlistlistingname{代码目录}
\renewcommand\lstlistoflistings{%
  \@centeredlisttitle{代\quad 码\quad 目\quad 录}{代\qquad 码\qquad 目\qquad 录}{lol}}

% 英文版（共享同一辅助文件）
\newcommand\listoffiguresEN{%
  \@centeredlisttitle{List of Figures}{List of Figures}{lof}}
\newcommand\listoftablesEN{%
  \@centeredlisttitle{List of Tables}{List of Tables}{lot}}
\newcommand\lstlistoflistingsEN{%
  \@centeredlisttitle{List of Listings}{List of Listings}{lol}}

\makeatother
